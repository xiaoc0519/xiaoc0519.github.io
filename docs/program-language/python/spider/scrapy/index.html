def get_cookies_dict():
    cookie_str = 'aa=mdmkdm;bb=sakdad;'
    cookies_dict = {}
    for item in cookie_str.split(';'):
        key, value = item.split('=',maxsplit=1)
        cookies_dict[key] = value
    return cookies_dict 

COOKIE_DICT = cookies_dict()    

def process_requests(self,request,spider):
