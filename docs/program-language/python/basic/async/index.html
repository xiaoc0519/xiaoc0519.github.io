    async & await 关键字
    在一个线程中遇到io等待时间，利用空闲的时间运行其他代码

    import asyncio, aiohttp

async def fetch(session, url): 
    async with session.get(url, verify_ssl = False) as response:
        content = await response.content.read() # 可等待对象
        file_name = url.rsplit('_')[-1]
        with open(file_name,'wb') as f:
            f.write(content)

async def main():
    async with aiohttp.ClientSession() as session:
        url_list = ['http:sadadasdada_sadadsa.jpg']
        tasks = [ async.creat_task( fetch(session,url)) for url in url_list ]
        await async.wait(tasks)

if __name__ == '__main__':
    asyncio.run( main() )

    asyncio 和 不支持异步的模块 # 

    import asyncio,requests

async def download(url):
    loop = asyncio.get_event_loop()
    future = loop.run_in_executor(None,requests.get,url)
    response = await future
    print('download success')
    with open('filename','w') as f:
        f.write(response.content)

if __name__ == '__main__':
    url_list = ['','','']
    tasks = [download(url) for url in url_list]
    asyncio.run(asyncio.wait(tasks))

    异步迭代器 # 

    class Reader(object):
    def __init__(self) -> None:
        self.count = 0

    async def readline(self):
        self.count += 1
        if self.count == 100:
            return None
        return self.count

    def __aiter__(self):
        return self

    async def __anext__(self):
        val = await self.readline()
        if val == None:
            raise StopAsyncIteration
        return val

async def func():
    obj = Reader()
    async for item in obj:
        print(item)

asyncio.run(func())

    异步操作redis # 

    # pip install aioredis
async def execute(add,pw):
    print('start')
    # 网络io操作 链接数据库
    redis = await aioredis.create_redis(add,password=pw)

    # 网络io操作 在redis中设置 哈希值 car， 内部再设三个兼职对
    # 即 redis = { car: {key1:1,key2:2,...}}
    await redis.hmset_dict('car',key=1,key2=2)

    # 网络io操作 去redis中获取值
    result = await redis.hgetall('car',encoding='utf-8')
    print(result)

    redis.close()
    # 网络io操作 关闭redis链接
    await redis.wait_clised()
    print('finish',add)

asyncio.run( execute('redis://0.0.0.0:00','root!12') )

import asyncio,aiomysql

async def execute():
    print('start')
    # 链接 mysql
    conn = await aiomysql.connect(host='',port='',user='',password='',db='')

    # 创建cursor
    cur = await conn.cursor()

    # 执行 sql
    await cur.execute('SELECT host,User FROM user')

    # 获取结果
    result = await cur.fetchall()
    print(result)

    # 关闭链接
    await cur.close()
    conn.close()

asyncio.run( execute() )
